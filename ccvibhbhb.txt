    <script>
        // ฟังก์ชันอัปเดตชื่อไฟล์เมื่อเลือกรูปภาพ
        document.getElementById('image_file').addEventListener('change', function(e) {
            if (e.target.files[0]) {
                document.getElementById('file-name-display').innerText = "เลือกไฟล์แล้ว: " + e.target.files[0].name;
            }
        });

        // ฟังก์ชันประมวลผลเพื่อแสดงผลบนหน้าจอเท่านั้น (ไม่ส่งไปที่ไหน)
        function processData() {
            let prompt = "=== [ข้อมูลจำลองเพื่อบันทึกลงระบบฐานข้อมูล] ===\n\n";
            let hasData = false;
            
            const age = document.getElementById('age').value;
            if(age) { prompt += `- อายุ: ${age} ปี\n`; hasData = true; }
            
            const gender = document.querySelector('input[name="gender"]:checked');
            if(gender) { prompt += `- เพศ: ${gender.value}\n`; hasData = true; }

            const weight = document.getElementById('weight').value;
            if(weight) { prompt += `- น้ำหนัก: ${weight} kg\n`; hasData = true; }

            const height = document.getElementById('height').value;
            if(height) { prompt += `- ส่วนสูง: ${height} cm\n`; hasData = true; }

            const underlying = document.getElementById('underlying').value.trim();
            if(underlying) { prompt += `- โรคประจำตัว: ${underlying}\n`; hasData = true; }

            const symptoms = document.getElementById('symptoms').value.trim();
            if(symptoms) { prompt += `- อาการผิวหนัง: ${symptoms}\n`; hasData = true; }

            const dist = document.querySelector('input[name="distribution"]:checked');
            if(dist) { prompt += `- ลักษณะผื่น: ${dist.value}\n`; hasData = true; }

            const sleep = document.querySelector('input[name="sleep"]:checked');
            if(sleep) { prompt += `- การนอนหลับ: ${sleep.value}\n`; hasData = true; }

            const file = document.getElementById('image_file').files[0];
            if(file) {
                prompt += `\n[รูปภาพแนบ]\n- ชื่อไฟล์: ${file.name} (พร้อมส่งไปคัดกรองตัวยาใน Supabase)\n`;
                hasData = true;
            }

            if(!hasData) {
                prompt += "(ยังไม่มีการกรอกข้อมูลใดๆ กรุณากรอกข้อมูลในฟอร์มเพื่อทดสอบระบบ)";
            }

            // แสดงกล่องผลลัพธ์และเลื่อนหน้าจอลงมาให้เห็น
            document.getElementById('promptOutput').textContent = prompt;
            document.getElementById('resultArea').style.display = 'block';
            document.getElementById('resultArea').scrollIntoView({ behavior: 'smooth' });
        }
    </script>
